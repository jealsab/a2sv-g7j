class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        freq = defaultdict(int)
        left = 0
        count = 0
        good_subarrays = 0

        for right in range(len(nums)):
            count += freq[nums[right]]
            freq[nums[right]] += 1

            while count >= k:
                good_subarrays += len(nums) - right
                freq[nums[left]] -= 1
                count -= freq[nums[left]]
                left += 1

        return good_subarrays
