class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        i = 0
        last_dup = 0
        ans = 1

        for j in range(1, len(s)):
            if s[j] == s[j - 1]:
                if last_dup != 0:
                    i = last_dup
                last_dup = j

            ans = max(ans, j - i + 1)

        return ans
