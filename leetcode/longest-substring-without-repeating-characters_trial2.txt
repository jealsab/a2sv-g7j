class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        i = 0
        maxLength = 0
        seen = {}

        for j in range(len(s)):
            if s[j] in seen and seen[s[j]] >= i:
                i = seen[s[j]] + 1

            seen[s[j]] = j
            maxLength = max(maxLength, j - i + 1)

        return maxLength
